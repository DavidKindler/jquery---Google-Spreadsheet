Example 2: Drop down lists and parent-sub filter set


$('#table2').GoogleSpreadsheet({
	'key': '0Aldl0jBslKHJdHdUR1VoNFNyUVBMR0luVTFUelRFRVE'   	
	, 'table_head' : '<thead><tr><th filter-type="ddl">Code</th><th >Title</th><th filter-type="ddl">
                   Description</th><th filter-type="ddl">Parent Filter</th><th filter-type="ddl">
                   Subfilter</th></tr></thead>'
  	, 'table_template': function(cell){return '<tr><td>'+cell[1]+'</td><td>'+cell[2]+'</td>
                       <td>'+cell[3]+'</td><td>'+cell[5]+'</td><td>'+cell[6]+'</td></tr>'}
	, 'subfilters' : { '4': '5'}				 
	})

Example 3: Parent-Sub filters, table class, alternate row classes, table head and template options set. Callback function assigned to remove waiting indicator.


$('#table3').GoogleSpreadsheet({
	'key': '0Aldl0jBslKHJdHdUR1VoNFNyUVBMR0luVTFUelRFRVE'	       
	, 'GS_Table_class' : 'Table3'
	, 'table_head' : '<thead><tr><th filter-type="ddl">Code</th><th >Title</th><th filter-type="ddl">
                     Description</th><th filter-type="ddl">Parent Filter</th><th filter-type="ddl">
                     Subfilter</th></tr></thead>'
  	, 'table_template': function(cell){if (cell[4]) return '<tr><td><a target="_new" href="'+
                      cell[4]+'">'+cell[1]+'</a></td><td>'+cell[2]+'</td><td>'+cell[3]+'</td>
                     <td>'+cell[5]+'</td><td>'+cell[6]+'</td></tr>'}	
	 , 'alternateRowClassNames' : ['evenRow','oddRow']
	 , 'subfilters' : { '4': '5'}		
	}, 	function(){ $('#waiting').hide()});
)

	})
